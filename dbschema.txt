CREATE TABLE BanList (
    BannedUserID      TEXT,
    BannedUserName    TEXT,
    ModeratorID       TEXT,
    ModeratorUserName TEXT,
    GuildID           TEXT,
    BanTime           INTEGER,
    ExpireTime        INTEGER,
    Reason            TEXT,
    Expires           BOOLEAN);

CREATE TABLE PrefixList (
    GuildID TEXT PRIMARY KEY
                 UNIQUE
                 NOT NULL,
    Prefix  TEXT NOT NULL
);

CREATE INDEX IF NOT EXISTS "" ON BanList (
    ModeratorID,
    GuildID,
    BannedUserID);
